import numpy as np

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    
    if np.random.choice([0,1], p = [1-epsilon, epsilon]): # 
        action = np.random.choice(actions) # 
    else: 
        action = max(qline,key=qline.get) #  Q 
    return action
